HTML

<ul class="tableview">
  <li class="tableview__item">リスト</li>
  <li class="tableview__item">
    <a href="#" class="tableview__target">
    リンク
    <span class="tableview__target-icon ico ico--arrow"></span>
    </a>
  </li>
  <li class="tableview__item">
    テキストの中に<a href="">インラインでリンク</a>が存在する場合
  </li>
</ul>

CSS

.tableview {
  overflow: hidden;
  border-radius: 8px;
  border: 1px solid #999;
}
.tableview__item {
  padding: 1em;
  background-color: #DDD;
}
.tableview__item + .tableview__item {
  border-top: 1px solid #666;
}
.tableview__target {
  position: relative; /* 矢印の配置の基点にする */
  padding-right: 2.4em; /* 矢印分の余白を空ける */
  display: block; /* インラインからブロックに変える */
  margin: -1em; /* paddingでふくらむ分を相殺する */
  padding: 1em; /* 必要な余白を取る */
  text-decoration: none; /* ``のデフォルトスタイルを消す */
  background-color: #DDF;
  color: inherit;
}
.tableview__target:hover {
  background-color: #E3E3FF;
}
.tableview__target-icon {
  position: absolute;
  top: 0; /* topとbottom、対するプロパティに値を持たせて、領域をつくる */
  bottom: 0; /* topとbottom、対するプロパティに値を持たせる、領域をつくる */
  right: 0.6em; /* アイコンが右端いっぱいに配置されないようにする */
  margin: auto; /* 垂直に中央になるようにする */
}

.ico {
  display: inline-block;
  background-image: url(spritesheets.png);
}
.ico--arrow {
  width: 16px;
  height: 16px;
  background-position: 0 -16px;
}
